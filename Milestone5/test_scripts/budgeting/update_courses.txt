-- Update offered courses at the end of a term
-- Existing row from dataset: WS1516 / 1051002-SWB / SWB / lecturer 86
UPDATE SCHEMA_MAIN.COURSE_ASSIGNMENT
SET TEACHING_HOURS = DECIMAL(3.50, 5, 2)
WHERE COURSE_ASSIGNMENT_TERM = 'WS1516'
  AND SUBJECT_NR = '1051002-SWB'
  AND DEGREE_SHORT_NAME = 'SWB'
  AND ACADEMIC_STAFF_ID = 86;
