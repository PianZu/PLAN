-- Check linked courses for not adding twice its workload
-- Schema has no "linked-course" attribute -> check duplicate assignments (double counting)
SELECT
  COURSE_ASSIGNMENT_TERM,
  SUBJECT_NR,
  DEGREE_SHORT_NAME,
  ACADEMIC_STAFF_ID,
  COUNT(*) AS CNT
FROM SCHEMA_MAIN.COURSE_ASSIGNMENT
GROUP BY COURSE_ASSIGNMENT_TERM, SUBJECT_NR, DEGREE_SHORT_NAME, ACADEMIC_STAFF_ID
HAVING COUNT(*) > 1;
