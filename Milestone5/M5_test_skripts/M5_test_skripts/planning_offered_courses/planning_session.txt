-- Start a planning session for the upcoming term
-- (implemented as a planning base view; schema has no "session" table)
CREATE OR REPLACE VIEW SCHEMA_MAIN.VW_PLANNING_BASE AS
SELECT
  c.DEGREE_SHORT_NAME,
  s.SUBJECT_SEMESTER_LVL,
  s.SUBJECT_NR,
  s.SUBJECT_NAME,
  s.SUBJECT_ELECTIVE,
  s.SUBJECT_CURRICULUM,
  s.SUBJECT_SCHEDULE
FROM SCHEMA_MAIN.COURSE c
JOIN SCHEMA_MAIN.SUBJECT s
  ON s.SUBJECT_NR = c.SUBJECT_NR
 AND s.DEGREE_SHORT_NAME = c.DEGREE_SHORT_NAME;
